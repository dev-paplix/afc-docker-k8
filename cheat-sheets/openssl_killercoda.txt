openssl genrsa -out ali_imran.key 2048

openssl req -new -key ali_imran.key -out ali_imran.csr -subj "/CN=ali_imran/O=dev/O=example.org"

openssl x509 -req -CA  /etc/kubernetes/pki/ca.crt -CAkey /etc/kubernetes/pki/ca.key -CAcreateserial -days 730 -in ali_imran.csr -out ali_imran.crt
kubectl config set-credentials ali_imran --client-certificate=ali_imran.crt --client-key=ali_imran.key
kubectl config set-context ali_imran-kubernetes --cluster=kubernetes --user=ali_imran --namespace=default
kubectl config get-contexts
